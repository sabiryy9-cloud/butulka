(+10 Xal)") xal_artir(user, 10) else: print(f"\n{Fore.RED}❌ YANLIŞ! O birini bəyəndi.") time.sleep(2) # --- Giriş Sistemi --- def daxil_ol(): os.system('cls' if os.name == 'nt' else 'clear') print("--- 🔑 DAXİL OL ---") ad = input("Adınız: ").strip() sifre = input("Şifrəniz: ").strip() db = melumatlari_yukle() if ad in db and db[ad]["sifre"] == sifre: return ad else: print("❌ Səhv ad və ya şifrə!") return None def qeydiyyat(): print("--- 📝 QEYDİYYAT ---") ad = input("Yeni Ad: ").strip() sifre = input("Yeni Şifrə: ").strip() db = melumatlari_yukle() if ad in db: print("❌ Bu ad artıq var!") else: db[ad] = {"sifre": sifre, "xal": 0} melumatlari_saxla(db) print("✅ Qeydiyyat uğurlu!") # --- Əsas Proqram --- if __name__ == "__main__": while True: print("\n1.
