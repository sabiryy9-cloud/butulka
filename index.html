import tkinter as tk from tkinter import messagebox import random import math from PIL import Image, ImageTk import pygame # Səs sistemi pygame.mixer.init() # Oyunçular players = [] angle = 0 spinning = False # UI root = tk.Tk() root.title("Spin the Bottle PRO+") root.geometry("500x700") root.config(bg="#1e1e1e") # Canvas canvas = tk.Canvas(root, width=400, height=400, bg="white", highlightthickness=0) canvas.pack(pady=10) # Butulka şəkli yüklə original_image = Image.open("assets/bottle.png") resized = original_image.resize((200, 200)) bottle_img = ImageTk.PhotoImage(resized) img_obj = canvas.create_image(200, 200, image=bottle_img) def rotate_image(a): rotated = original_image.rotate(a) resized = rotated.resize((200, 200)) return ImageTk.PhotoImage(resized) def add_player(): name = entry.get() if name: players.append(name) listbox.insert(tk.END, name) entry.delete(0, tk.END) def remove_player(): selected = listbox.curselection() if selected: idx = selected[0] listbox.delete(idx) players.pop(idx) def spin(): global spinning, angle if spinning or len(players) < 2: messagebox.showwarning("Xəta", "Ən azı 2 oyunçu lazımdır!") return spinning = True angle = 0 pygame.mixer.Sound("assets/spin.wav").play() rotate() def rotate(): global angle, spinning, bottle_img if angle < 360 * random.randint(5, 8): angle += random.randint(15, 25) bottle_img = rotate_image(angle) canvas.itemconfig(img_obj, image=bottle_img) root.after(30, rotate) else: spinning = False select_players() def select_players(): p1, p2 = random.sample(players, 2) mode = random.choice(["Truth", "Dare"]) result_label.config( text=f"{p1} 👉 {mode} seçir\n{p2} 👉 cavab verir" ) # UI elementləri frame = tk.Frame(root, bg="#1e1e1e") frame.pack() entry = tk.Entry(frame, font=("Arial", 12)) entry.pack(side=tk.LEFT, padx=5) add_btn = tk.Button(frame, text="➕", command=add_player) add_btn.pack(side=tk.LEFT) remove_btn = tk.Button(frame, text="❌", command=remove_player) remove_btn.pack(side=tk.LEFT) listbox = tk.Listbox(root, height=6) listbox.pack(pady=10) spin_btn = tk.Button(root, text="🎯 FIRLAT", command=spin, bg="green", fg="white", font=("Arial", 16)) spin_btn.pack(pady=15) result_label = tk.Label(root, text="", font=("Arial", 14), fg="white", bg="#1e1e1e") result_label.pack(pady=10) root.mainloop()
